`src/utils/keys.ts`
Parses the private key into a Solana Keypair.
import { Keypair } from '@solana/web3.js';
import bs58 from 'bs58';

export function getKeypair(privateKey: string): Keypair {
  try {
    const secretKey = bs58.decode(privateKey);
    return Keypair.fromSecretKey(secretKey);
  } catch {
    const secretKeyArray = JSON.parse(privateKey);
    if (Array.isArray(secretKeyArray) && secretKeyArray.length === 64) {
      return Keypair.fromSecretKey(Uint8Array.from(secretKeyArray));
    }
    throw new Error('Invalid private key format');
  }
}

`src/swap.ts`
Handles swap logic with Jupiter Swap v6.
import { Connection, Keypair } from '@solana/web3.js';
import { Jupiter } from '@jup-ag/core';

export async function performSwap(
  connection: Connection,
  wallet: Keypair,
  buyTokenMint: string,
  sellTokenMint: string,
  amount: number
): Promise {
  const jupiter = await Jupiter.load({
    connection,
    cluster: 'mainnet-beta',
    user: wallet,
    version: 6, // Jupiter v6 API
  });

  const amountLamports = Math.floor(amount * 1e9); // Adjust decimals as needed
  const routes = await jupiter.computeRoutes({
    inputMint: sellTokenMint,
    outputMint: buyTokenMint,
    amount: amountLamports,
    slippageBps: 50, // 0.5% slippage
  });

  const bestRoute = routes.routesInfos[0];
  const { execute } = await jupiter.exchange({ routeInfo: bestRoute });
  await execute();
  console.log(`Swap executed: ${amount} ${sellTokenMint} -> ${buyTokenMint}`);
}
Notes: Assumes Jupiter v6 API is compatible with this structure; adjust if the API changes.

`src/dca.ts`
Manages DCA strategies with a single timer.
import { Connection, Keypair } from '@solana/web3.js';
import strategies from './config/strategies';
import { performSwap } from './swap';

interface StrategyState {
  config: typeof strategies[0];
  nextExecutionTime: number;
  buysCompleted: number;
}

export async function runDCA(connection: Connection, wallet: Keypair): Promise {
  const strategyStates: StrategyState[] = strategies.map((s) => ({
    config: s,
    nextExecutionTime: Date.now() + s.dca.intervalMinutes * 60 * 1000,
    buysCompleted: 0,
  }));

  async function scheduleNextBuy(): Promise {
    const activeStrategies = strategyStates.filter(
      (s) => !s.config.dca.totalBuys || s.buysCompleted < s.config.dca.totalBuys
    );
    if (activeStrategies.length === 0) {
      console.log('All DCA strategies completed');
      return;
    }

    const nextStrategy = activeStrategies.reduce((a, b) =>
      a.nextExecutionTime < b.nextExecutionTime ? a : b
    );
    const delay = Math.max(0, nextStrategy.nextExecutionTime - Date.now());

    setTimeout(async () => {
      try {
        await performSwap(
          connection,
          wallet,
          nextStrategy.config.swap.buyTokenMint,
          nextStrategy.config.swap.sellTokenMint,
          nextStrategy.config.dca.amount
        );
        nextStrategy.buysCompleted++;
        nextStrategy.nextExecutionTime = Date.now() + nextStrategy.config.dca.intervalMinutes * 60 * 1000;
      } catch (error) {
        console.error('Swap failed:', error);
      }
      scheduleNextBuy();
    }, delay);
  }

  console.log('Starting DCA bot...');
  await scheduleNextBuy();
}
